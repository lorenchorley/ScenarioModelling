Entity Actor {
    State Bob
    CharacterStyle "Red"
}

SM Name {
    Bob -> Alice : ChangeName
    Alice -> Bob : ChangeName
}

Scenario First {
    Dialog {
        Character Actor
        Text "Hello"
    }
    Dialog SayName {
        Character Actor
        Text "My name is {Actor.State}"
    }
    Choose {
        Change "Change name and repeat"
        GoodBye "Ciao"
    }
    
    Transition Change {
        Actor : ChangeName
    }
    if <Actor.State == "Alice"> {
        Dialog {
            Character E1
            Text "I am now Alice ! "
        }
    }
    Jump {
        SayName
    }
    Dialog GoodBye {
        Text "Bubye (Actor called {Actor.State} in the end)"
    }
}
